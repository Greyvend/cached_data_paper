\begin{theorem}
$P^{\ast} \subseteq \pi_{X^{\ast}} ( \sigma_{F^{\ast}[X]} (P_1 \Join \dots \Join
P_n))$, where $X = \bigcupn X_{v}$ if:
\\a) $X^{\ast} \subseteq X$
\\b)
$ \bigcupn \{R^{v}_{1}, \ldots, R^{v}_{s(v)}\} = \{R'_{1}, \ldots, R'_{s'}\}
\subseteq
\{R^{\ast}_{1}, \ldots, R^{\ast}_{l}\} $
\\c) $M(F^{\ast}) \subseteq M(F_{v}), v = 1,\dots,n $.

\label{th_mult}
\end{theorem}
