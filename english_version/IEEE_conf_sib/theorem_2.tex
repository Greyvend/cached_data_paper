\begin{theorem}
$P^{\ast} = \pi_{X^{\ast}} (\sigma_{F^{\ast}} (P_{v}))$, if:
\\a) $X^{\ast} \subseteq X_{v}$
\\b) $\{R^{v}_{1}, \ldots, R^{v}_{s(v)}\} = \{R^{\ast}_{1}, \ldots,
R^{\ast}_{l}\}$
\\c) $M (F^{\ast}) \subseteq M (F_{v}) $
\\d) $ \<F^{\ast}> \subseteq X_{v} $.
\label{th_base_eq}
\end{theorem}
